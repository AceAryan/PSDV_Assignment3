import pymc as pm  # Import PyMC for probabilistic modeling
import numpy as np  # Import NumPy for numerical computations
import matplotlib.pyplot as plt  # Import Matplotlib for visualization

# Define parameters for two different Binomial distributions
n1, p1 = 20, 0.3  # First distribution: 20 trials, 30% success probability
n2, p2 = 8, 0.7   # Second distribution: 8 trials, 70% success probability

# Define the probabilistic model
with pm.Model():
    # Define two Binomial distributions
    binom1 = pm.Binomial("binom1", n=n1, p=p1)  # First Binomial variable
    binom2 = pm.Binomial("binom2", n=n2, p=p2)  # Second Binomial variable
    
    # Sample from the prior predictive distribution (no data yet, just theoretical sampling)
    trace = pm.sample_prior_predictive(samples=10_000)

# Extract the generated samples from the prior predictive trace
samples1 = trace.prior["binom1"]
samples2 = trace.prior["binom2"]

# Compute unique values (possible outcomes) and their frequencies for each distribution
x1, counts1 = np.unique(samples1, return_counts=True)
x2, counts2 = np.unique(samples2, return_counts=True)

# Normalize the frequencies to get probabilities (PMF values)
y1 = counts1 / np.sum(counts1)
y2 = counts2 / np.sum(counts2)

# Plot the probability mass function (PMF) for both distributions
plt.plot(x1, y1, 'bo-', label=f'n = {n1}, p = {p1}')  # Blue circles for first distribution
plt.plot(x2, y2, 'o-', color='orange', label=f'n = {n2}, p = {p2}')  # Orange circles for second distribution

# Label the axes
plt.xlabel('x')  # X-axis represents the number of successes
plt.ylabel('pmf(x)')  # Y-axis represents the probability mass function

# Add a legend to differentiate the two distributions
plt.legend()

# Display the plot
plt.show()

Sampling: [binom1, binom2]

2.Gaussian/Normal Distribution¶

In [20]:
import pymc as pm  # Import PyMC for probabilistic modeling
import matplotlib.pyplot as plt  # Import Matplotlib for plotting
import seaborn as sns  # Import Seaborn for smooth KDE plots

# Define a probabilistic model
with pm.Model():
    # Define a Normal (Gaussian) distribution with mean 0 and standard deviation 1
    normal_dist = pm.Normal("normal_dist", mu=0, sigma=1)
    
    # Sample 1000 values from the posterior distribution
    trace = pm.sample(1000, progressbar=False)

# Extract the sampled values from the posterior distribution
samples = trace.posterior["normal_dist"].values.flatten()

# Create a figure for the plot
plt.figure(figsize=(8, 5))

# Plot the density estimation using Seaborn
sns.kdeplot(samples, fill=True, color="royalblue", linewidth=2)

# Set title and labels for clarity
plt.title("Normal Distribution (μ=0, σ=1)", fontsize=14)
plt.xlabel("Value", fontsize=12)
plt.ylabel("Density", fontsize=12)

# Add a grid for better readability
plt.grid(True, linestyle="--", alpha=0.5)

# Display the plot
plt.show()
